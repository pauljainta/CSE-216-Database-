SELECT NVL(DEPARTMENT_ID,-1),AVG(NVL(COMMISSION_PCT,0))
FROM EMPLOYEES
GROUP BY DEPARTMENT_ID;

SELECT E.EMPLOYEE_ID, J.JOB_ID,J.JOB_TITLE
FROM EMPLOYEES E JOIN JOBS J ON (E.JOB_ID=J.JOB_ID AND E.SALARY>10000) JOIN JOB_HISTORY JH ON(E.EMPLOYEE_ID=JH.EMPLOYEE_ID AND JH.START_DATE<'12-DECEMBER-2003');

SELECT DEPARTMENT_ID,MAX(SALARY) MAX_SALARY 
FROM EMPLOYEES
GROUP BY DEPARTMENT_ID
HAVING MIN(SALARY)>8000
ORDER BY MAX(SALARY);
